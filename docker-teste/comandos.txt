Ao realizar uma conexão usando o Express deve setar no app.listen(3000, '0.0.0.0');

# -t serve para dar um nome ao nosso container
# . serve para indicar onde está o Dockerfile, no caso está no diretório atual
# Agora irá rodar tudo do Dockerfile, irá montar a imagem
docker build -t mayderson/fullstackapp .


# Rodar a aplicação
# -p porta que irá acessar primeiro porta do navegador e depois porta que o container irá acessar
# -d qual será a imagem que irá usarmos
docker run -p 3000:3000 -d mayderson/fullstackapp

# lista as máquinas que estão rodando
docker ps

# Iniciar
# --build para montar a imagem caso não tenha montado
docker-compose up


# Stop no container
docker stop CONTAINER ID

# Remove no container
docker rm CONTAINER ID
